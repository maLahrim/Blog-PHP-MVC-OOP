# Page Principales 

RewriteRule ^accueil$   [L]
RewriteRule ^admin$  action=admin [L]
RewriteRule ^chapitres$  action=posts [L]

#Font-End
RewriteRule ^chapitre-([0-9]+)$ action=chapitre&id=$1 [L]

#Back-End

#Edition de chapitre 

RewriteRule ^admineditionchapitre-([0-9]+)$ action=admin&chapitre=$1 [L]

#Supression de chapitre 

RewriteRule ^adminsupressionchapitre-([0-9]+)$ action=admin&deletePost=$1 [L]

#Supression de commentaire

RewriteRule ^adminsupressioncommentaire-([0-9]+)$ action=admin&deleteComment=$1 [L]

#Nouveau post

RewriteRule ^adminnouveau-chapitre$ action=admin&newPost [L]
RewriteRule ^adminpublier$  action=admin&insertPost [L]
